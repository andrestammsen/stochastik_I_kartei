\begin{karte}[Stochastik-I]	{Eigenschaften des Maßintegrals}[prüfungsrelevant]
$ f,g \in \overline{Z\XcalA} \ \textrm{und} \ \XcalAmue $ Maßraum, dg:
\begin{enumerate}[label=(\roman*)]
	\item $ f = 0 \ \mueFastSicher \Longrightarrow \integral{}{}{f}{\mu} = 0 $
	\item $ \integral{}{}{f}{\mu} = 0 \Longleftrightarrow f = 0 \ \mueFastSicher \
	\textrm{(falls} \ f \geq 0 \textrm{)} $
	\item $ \integral{}{}{g}{\mu} \ \textrm{existiert und} \ \integral{}{}{g}{\mu} = %\integral{}{}{f}{\mu} \ \Longleftarrow \vspace{0.15 cm} \\
	\textrm{(falls} \ \integral{}{}{f}{\mu} \ \textrm{existiert und} \ f = g \ \mueFastSicher \textrm{)} $
	\item $ f \ \mueIntegrierbar \  \Longrightarrow f \ \textrm{ist} \ \mueFastSicher \ \textrm{endlich} $
\end{enumerate}
\end{karte}
