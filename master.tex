\begin{karte}[Stochastik-I]{Algebra \\ $ \sigma $-Algebra \\ Dynkin-System}[2.1.2]
Ein System von Teilmengen/Ereignissen einer beliebigen, nicht-leeren Menge $ X $ heißt \textbf{Algebra} über $ X $, falls
\begin{enumerate}[label=(\roman*), noitemsep]
\item $ X \in \calA $
\item $ A \in \calA \Longrightarrow \comp{A} \in \calA $
\item $ A_1, ... , A_n \in \calA \Longrightarrow \cup_{i=1}^{n} A_i \in \calA $ (n $ \geq $ 1 beliebig)
\item $ A_1, A_2, ...  \in \calA \Longrightarrow \cup_{i=1}^{\infty} A_i \in \calA \ \longrightarrow $ \bm{$\sigma$}\textbf{-Algebra}
\item $ A_1, A_2, ... p.d.\Longrightarrow \cup_{i=1}^{\infty} A_i \in \calA \ \longrightarrow $ \textbf{Dynkin-System}
\end{enumerate}
\end{karte}

\begin{karte}[Stochastik-I]{Metrik \\ und \\ metrischer \\ Raum}[2.1.6]
Metrischer Raum ist bel n-leere Menge $X$ versehen mit Metrik (Abstand) $d$, wobei $d$ Abbildung vom Typ \cmhalf \\
\centering{$d: X \times X \ni (x,y) \longrightarrow d(x,y) \in \calR_{\geq 0}$} \cmhalf \\
\raggedright{mit den Eigenschaften} \mmthree
\begin{enumerate}[label=(\roman*), noitemsep]
\item $d(x,y) = 0 \Leftrightarrow x = y$ \ (Definitheit)
\item $d(x,y) = d(y,x) \ \forall x,y \ni X$ \ (Symmetrie)
\item $d/x,z) \leq d(x,y) \, + \, d(y,z) \ \forall x,y,z \in X$ \ ($\Delta$-Ungl)
\end{enumerate}
\end{karte}

\centering{$K(x,\eps) := {y \ni X : d(x,y) < \eps}$ \mmthree \\}

\begin{karte}[Stochastik-I]{additiv \\ und \\ $sigma$-additiv}[zu 2.2.1]
\raggedright {

$$

\begin{karte}[Stochastik-I]{nicht-negative \\ Mengenfunktion \\ (finiter) Maßraum \\ Wahrscheinlichkeitsraum}[2.2.1]
Seien $X$ n-leere Menge, $\emptyset \in \calA \subset \calP(X)$, dann ist $(X, \calA, \mu)$ eine n-neg Mengenfunktion auf $(X, \calA)$, wenn $\mu(\emptyset) = 0$ und: \newline
$\mu: \calA \ni A \longrightarrow \mu(A) \in [0,\infty]$
\begin{enumerate}[label=(\roman*), nosep]
\item
$\mu$ additiv $ \longrightarrow (X, \calA, \mu)$ ist \textbf{Inhalt}
\item
$\mu \ \sigma$-additiv $ \longrightarrow \mu$ ist \textbf{Maß}, $(X, \calA, \mu)$ ist \textbf{Maßraum}
\item
$\mu(X) < \infty \longrightarrow (X, \calA, \mu)$ ist \textbf{finiter Maßraum} \\
$\mu(X) = 1 \longrightarrow  (X, \calA, \mu)$ ist  \textbf{Wahrsch-Raum}
\item
$\exists \, A_1, A_2, ... \in \calA, \mu(A_i) < \infty, \bcupp{i=1}{\infty}{A_i} = X \longrightarrow (X, \calA, \mu) $ ist \bm{$\sigma$}\textbf{-finiter Maßraum}
\end{enumerate}
\end{karte}

\begin{karte}[Stochastik-I]{numerische \\ Abbildungen}[2.3.3]

\begin{karte}[Stochastik-I]{Maßintegral \\ Schreibweisen}[unter 2.4]
Folgende Schreibweisen sind für das Maßintegral gebräuchlich:
$$
\mu(f) \equiv \int_{}^{}f \, \t{d}\mu \equiv \int_{}^{} f(x)\mu(\t{d}x) \equiv \int_{X}{} f \, \t{d}\mu \equiv \int_{y}{} f(x) \mu(\t{d}x) 
$$
Bezogen auf eine Menge $A \in \calA$ schreibt man:
$$
\int_{A}{} f(x)\mu(\t{d}x) \equiv \int_{A}{} f \, \t{d}\mu \equiv \mu(f;A) \equiv \mu(f \cdot I_A) \equiv \int_{X}{} f \, I_A \, \t{d}\mu
$$
\end{karte}

\begin{karte}[Stochastik-I]{Maßintegral \\ nicht-negativer \\ messbarer \\ Funktionen}[2.4.6]

\begin{karte}[Stochastik-I]{Positivteil \& Negativteil \\ von \\ Funktionen}[2.4.8]

f^+(x) := max\{f(x),0\}, \ f^-(x) := max\{-f(x),0\}, \ x \in X
$$
\end{karte}

\begin{karte}[Stochastik-I]{Maßintegral \\ von $f$ bzgl $\mu$ \\ oder \\ $\mu$-Integral von f}[2.4.9]
$f \in \overline{Z(X,\calA)}$ bel, $f^+ / f^-$ Positiv-/Negativteil von $f$:
\begin{enumerate}[label=(\roman*), noitemsep] 
\item Mind $\mu(f^+) \t{ oder } \mu(f^-)$ endl, dann ist das 
\textbf{Maßintegral von $\bm{f}$ bzgl $\bm{\mu}$} bzw \textbf{$\bm{\mu}$-Integral von $\bm{f}$}:
$$
\mu(f) \equiv \int_{}^{}f \dd\mu := \int_{}^{} f^+ \dd\mu - \int_{}^{} f^- \dd\mu \equiv \mu(f^+) - \mu(f^-)
$$
\item Integral exist: $f$ heißt \textbf{$\bm{\mu}$-quasiintegrierbar}
\item Integral exist und endl: $f$ ist \textbf{$\mu$-integrierbar}
\item $\mu(f^+) = \infty \t{ und } \mu(f^-) = \infty \t{: } \mu(f)$ nicht definiert
\end{enumerate}
\end{karte}

bzw $ S_\eta = (b_i)_{i \in \N} \subset \R $ dann ist auch $ \varphi $ diskret mit mögl Sprungstellen $ c_k = a_i + b_j $ und es gilt $ \forall x \in \R $:
\begin{align*}
\PP(\varphi \leq x)
= \summ{i \geq 1}{}{ \PP_\xi(\{a_i\}) \PP_\eta(\{c_k-a_i\})} \\
= \summ{i \geq 1}{}{ \PP_\eta(\{b_i\}) \PP_\xi(\{c_k-b_j\})}
\end{align*}
\end{karte}

\begin{karte}[Stochastik-I]{Bernoulli- \\ Verteilung}[3.3.1]
\vspace*{1 cm}
Eine ZV besitzt eine \textbf{Bernoulli-Verteilung} zum \newline
Erfolgsparameter $0\leq p \leq 1$, kurz $\xi \sim BERN(p)$, \newline
wenn $S = \{0,1\}$ und $\PP_\xi(\{1\}) = p$
\vspace*{0.3 cm} \\
\end{karte}

\begin{karte}[Stochastik-I]{Binomial- \\ Verteilung}[3.3.1]
Eine ZV $\xi$ besitzt eine \textbf{Binomial-Verteilung} zu \newline
den Parametern $n,p$ mit $n \in \N$ und $0 < p < 1$, \newline
kurz $\xi \sim BINOM(n,p)$, wenn $S = {0,1,...,n}$, $\xi = \sum_{i=1}^{n} \eta_i$: \cmhalf \\
$$
\PP_\xi(\{k\}) = \binom{n}{k}p^k(1 - p)^{n-k} \quad \forall k \in S
$$ \cmhalf \\
\textit{Die binomialverteilte ZV entsteht also als Summe IID. Bernoulli Variablen}
\end{karte}

\begin{karte}[Stochastik-I]{Grenzwertsatz \\ von \\ Poisson}[3.3.3]
Sei $(\xi_n)_{n \in \N}$ eine Folge von binomial-verteilten ZV über $(\Omega, \calA, \PP)$ mit $\xi_n \sim BINOM(n,p_n)$ für $n \in \N$ und $n \, p_n = \lambda > 0 \ \forall n \in \N$, dg:
\vspace*{0.3 cm} \\
$$
\lim_{n \to \infty} \PP(\xi_n = k) = e^{-\lambda} \left( \frac{\lambda^k}{k!} \right) \quad \forall k \geq 0
$$
\vspace*{0.3 cm} \\
\textit{(Beispielsweise radioaktiver Zerfall; extrem große Menge an Experimenten benötigt.)}
\end{karte}

\begin{karte}[Stochastik-I]{Poisson- \\ Verteilung}[3.3.4]
Eine ZV $\xi$ besitzt eine \textbf{Poisson-Verteilung } zum \newline
Parameter $\lambda > 0$, kurz $\xi \sim POIS(\lambda)$, wenn $S = \N_0$ und \mmthree \newline
$$
\PP_\xi (\{k\}) = \frac{\lambda^k}{k!} \, e^{(-\lambda)}
$$ \mmthree \newline
\textit{(Große Population/seltenes Merkmal: zugrunde liegende Binomial-Vert durch Poisson-Vert ersetzen mit $\lambda = np$, n = Populationsgröße, $p$ = W., dass Individuum Merkmal trägt; z.B. Fälle seltener Krankheit in großer Stadt/Jahr.)}
\end{karte}

\begin{karte}[Stochastik-I]{Multinomial- \\ Koeffizient}[3.3.3]
Für $k_1, ... , k_m$ mit $\sum_{i=1}^{m} k_i = n$ definiert durch
$$
\binom{n}{k_1,k_2,...,k_m} := \frac{n!}{k_1! \cdot k_2! \cdot \cdot \cdot k_m!}
$$
$$
\varphi := \left( \summ{i=1}{n}{I(\eta_i = 1)}, \, \summ{i=1}{n}{I(\eta_i = 1)}, ... \, , \, \summ{i=1}{n}{I(\eta_i = 1)} \right) ^T
$$
$$
\PP(\varphi = (k_1,k_2, ... \, k_m)^T) = \binom{n}{k_1,k_2, ... \, k_m} p_1^{k_1} \cdot p_2^{k_2} \cdot \cdot \cdot p_m^{k_m} 
$$
\end{karte}

\begin{karte}[Stochastik-I]{Multinomial- \\ Verteilung}[3.3.6]
Bei einer \textbf{Multinomial-Verteilung} $\varphi$ zu Parametern $(n,p_1,p_2, ... \, p_m)$, kurz $\varphi \sim MULTINOM(n,p_1,p_2, ... \, p_m)$ kann der zugehörige diskrete ZV nur folgende Werte annehmen
$$
S = \{ (k_1,k_2, ... \, k_m)^T: \summ{i=1}{m}{k_i = n}, \, k_i \in \N_0, \, 1 \leq i \leq m \}
$$
\textit{(Dies ist eine Verallgemeinerung der Binomial-Verteilung, bei der die $n$ Objekte nicht auf zwei, sonderm auf $m$ Gruppen verteilt werden.)}
\end{karte}

\begin{karte}[Stochastik-I]{Geometrische \\ Verteilung}[3.3.7]
ZV $\xi$ hat \textbf{geometrische Verteilung} zum Parameter $ 0 < p < 1$, kurz $\xi \sim GEOM(p)$, falls $S = N_0$ und $ \forall k \in S$:
$$
\PP_\xi(\{k\}) = (1 - p)^k p
$$
\textit{(z.B. Wartezeit eines (nicht lernenden) Schützen bis zum ersten Treffer ($\xi = k$: der Schütze trifft beim Schuss $k + 1$)}
\end{karte}

\begin{karte}[Stochastik-I]{Hypergeometrische \\ Verteilung}[3.3.8]
ZV $\xi$ hat \textbf{Hypergeometrische Verteilung} zum  \newline
Parameter $(n,r,s)$ mit $n,r,s \in \N$ un $n \leq r$, \newline
kurz $\xi \sim HYPER(n,r,s)$, falls $S = {0,1, ... \, n}$ und $\forall k \in S$:
$$
\PP_\xi(\{k\}) = \frac{\binom{r}{k} \binom{s}{n-k}}{\binom{r+s}{n}}
$$
\textit{(Urne mit $r$ roten und $s$ schwarzen Kugeln, dann gibt die ZV $\xi \sim HYPER(n,r,s)$ die Anzahl gezogener roter \newline
Kugeln beim $n$-maligen Ziehen ohne Zurücklegen an.)}
\end{karte}

\begin{karte}[Stochastik-I]{Jacobi- \\ Matrix}[3.3.9]
$T: \R^n \ni x \longrightarrow T(x) \in \R^n$ diff-bare Abb mit stet Abl gdw $\forall T_i: T_i$ ist partiell stet diff-bar mit \newline
$D_j T_i(x) = \frac{\partial T_i (x)}{\partial x_j}, \, 1 \leq i \leq n$,
dann ist $T'(x)$ die \textbf{Jacobi-Matrix} von $T$ an der Stelle $x$, dh
\mmthree
$$
T'(x) := 
\left( \begin{matrix}
D_1 T_1 (x) & D_2 T_1 (x) & \dots & D_1 T_n (x) \\
D_1 T_2 (x) & D_2 T_2 (x) & \dots & D_2 T_n (x) \\
\vdots      & \vdots      & \vdots & \vdots      \\
D_1 T_n (x) & D_2 T_n (x) & \dots & D_n T_n (x) \\
\end{matrix} \right )
$$
\end{karte}

\begin{karte}[Stochastik-I]{Transformations- \\ Formel}[3.3.10]

\begin{enumerate}[label=(\roman*), noitemsep]
\item $T(x)$ hat stet part Abl an jeder Stelle $x \in \R^n$
\item $T$ ist bij, $T^{-1}$ hat stet part Abl an jeder Stelle $y \in \R^n$
\end{enumerate}
Dann hat das Bildmaß von $\lambda^n$ unter $T$, dh \mmthree \newline
$\lambda_T^n \equiv (\lambda^n)_T: \calB_n^* \ni B \longrightarrow \lambda_t^n(B): \lambda^n(T^{-1}(B)) \in \R_\geq$  \mmthree \newline
eine Radon-Nikodym Dichte $f$ bzgl $\lambda^n$, dh d$\lambda_T^n = f$ d$\lambda^n$:  \mmthree \newline
$f(y) = | \, det((T^{-1}) ' (y))|$
\end{karte}

\begin{karte}[Stochastik-I]{Kovarianz \\ und \\ Varianz-Kovarianz}[3.6.1]

f(x;\mu,\sigma^2) := \frac{1}{\sqrt{2 \pi \sigma^2}} \ \t{exp}\left( - \frac{(x - \mu)^2}{2 \sigma^2} \right) \in \R_{\geq} \\
$$
\end{karte}

\begin{karte}[Stochastik-I]{Standard- \\ Normalverteilung}[3.3.13]
Eine \tb{Standard-Normalverteilung} liegt vor wenn $\newline
\mu = 0, \, \sigma^2 = 1$, also $\xi \sim \calN(0,1)$. Dichte und Vert-Fkt:
\cmhalf \\
$$
\phi: \R \ni x \longrightarrow \phi(x) := \frac{1}{\sqrt{2 \pi}} \t{exp} \left( - \frac{x^2}{2} \right) \in \R_\geq
$$
\mmone \\
$$
\Phi: \R \ni x \Longrightarrow \phi(x) := \int_{-\infty}^{x} \phi(tI \lambda(\dd t) \in [0,1]
$$
\end{karte}

\begin{karte}[Stochastik-I]{Informationen \\ zur \\ Normalverteilung}[3.3.14]
Seien $\xi \sim \calN(\mu,\sigma^2) $ und $\eta \sim \calN(0,1)$:
\begin{enumerate}[label=\roman*), noitemsep]
\item Dichte $f_{\mu,\sigma^2} $ von $\xi$ ist symm um $\mu$, Maximum im Symm-Zentrum $\mu$ - dem \tb{Lokationsparameter} - und Wendepunkte bei $\mu \pm \sigma$.
\item Variable $\sigma \eta + \mu$ hat $\calN(\mu,\sigma^2)$-Verteilung
\item Variable $(\xi - \mu) / \sigma$ hat $\calN(0,1)$-Vert. Diese Transformation nenn man \tb{Standardisieren}.
\item Lesen der Tafel mit $\Phi(x) = 1 - \Phi(-x)$.
\end{enumerate}
\end{karte}

\begin{karte}[Stochastik-I]{Normalverteilungsfamilie \\ abgeschlossen \\ unter \\ Faltung}[3.3.15]
Seien $\xi_1,\xi_2$ unabh Z-Variablen mit $\xi_1 \sim \calN(\mu_1,\sigma_1^2)$ und $\xi_2 \sim \calN(\mu_2,\sigma_2^2)$, dg:
\cmonehalf \newline
$$
\xi_1 + \xi_2 \sim \calN(\mu_1 + \mu_2, \, \sigma_1^2 + \sigma_2^2)
$$
\end{karte}

\begin{karte}[Stochastik-I]{Gamma-Verteilung}[3.3.16]
\raggedright{Z-Variable $\xi$ hat \tb{Gamma-Verteilung} zu Parametern $p > 0, \ \lambda > 0$, kurz $\xi \sim \Gamma(p,\lambda)$, falls $\PP_\xi$ abs-stet bzgl $\lambda$ mit Dichte-Fkt und \tb{Eulers Gamma-Funktion}: \mmthree \\}
\centering{$\gamma_{p,\lambda} \equiv \gamma : \R_> \ni x \rightarrow \gamma_{p,\lambda}(x) \equiv \gamma(x; \, p,\lambda) $}
$$
\gamma(x; \, p,\lambda) := \frac{\lambda^p x^{p-1} \t{exp}(-\lambda x)}{\Gamma(p)} \in \R_\geq
$$
$$
\Gamma(p) := \int_{0}^{\infty}t^{p-1}\t{exp}(-1)\dd t
$$
%\end{gather*}
\end{karte}

\begin{karte}[Stochastik-I]{Exponential- \\ Verteilung}[3.3.17]
Z-Variable $\xi$ gamma-vert mit Parametern $p = 1, \lambda > 0$, also $\xi \sim \Gamma(1,\lambda)$ heißt \tb{exponential-verteilt} zum \newline
Parameter $\lambda$, kurz $\xi \sim EXP(\lambda)$. Durch Einsetzen von \newline
$p = 1$ ergibt sich aus der allg Formel:
\cmhalf \\
$$
\gamma(x; 1, \lambda) := \lambda \t{exp}(-\lambda x)
$$
\end{karte}

$\lambda = 1/2$ und $\xi \sim \Gamma(k/2, 1/2),\lambda)$ heißt \tb{chiquadrat-verteilt} mit $k$ \tb{Freiheitsgraden}, kurz $\xi \sim \chi_k^2$. %Durch Einsetzen von $p = 1$ ergibt sich aus der allg Formel:

$$
\end{karte}

\begin{karte}[Stochastik-I]{Eigenschaften \\ Gamma-Verteilung}[3.3.18]
Eigenschaften der Gamma-Verteilung:
\begin{enumerate}[label=(\roman*)]
\item $\forall p > 0:\gamma(p+1) = p\gamma(p)$ \qquad (mittels part. Int.)
\item $\xi \sim \gamma(p,\lambda)$ gdw $\lambda \cdot \sim \gamma(p,1)$, \newline
dh $1/\lambda$ ist ein Skalenparameter
\item Modellierung n-neg Ereignisse, z.B. Vert-Modell \newline
Lebensdauer oder räumliche Abst Ursache/Wirkung
\end{enumerate}
\end{karte}

\begin{karte}[Stochastik-I]{Beta- \\ Verteilung}[3.3.19]
\raggedright{Z-Variable $\xi$ hat \tb{Beta-Verteilung} zu Parametern $r > 0, \ s > 0$, kurz $\xi \sim B(r,s)$, falls $\PP_\xi$ abs-stet bzgl $\lambda$ mit Dichte-Fkt und \tb{Beta-Funktion}: \\}
\centering{$ \beta_{r,s} \equiv \beta : ]0,1] \ni x \longrightarrow \beta_{r,s} \equiv \beta(x; r,s) $}
$$
\beta(x; r,s) = \frac {x^{r-1} (1-x)^{s-1}}{B(r,s)} \in \R_\geq
$$
$$
B(r,s) :) \frac{\Gamma(r)\Gamma(s)}{\Gamma(r+s)}
$$
\end{karte}

\begin{karte}[Stochastik-I]{Zusammenhang \\ unabhängige \\ gamma-verteilte \\ Zufallsvariablen}[3.3.20]
Für $p > 0, \ q > 0, \ \lambda > 0$ seien $\xi_1 \sim \Gamma(p,\lambda)$ und  $\xi_2 \sim \Gamma(q,\lambda)$ unabh Z-Variable, dann folgt:
$$
\eta_1 := \xi_1 + \xi_2 \t{ und } \eta_2 := \frac{\xi_1}{\xi_1 + \xi_2}
$$
sind ebenfalls wieder unabh mit
$$
\eta_1 \sim \Gamma(p+q,\lambda) \t{ und } \eta_2 \sim B(p,q)
$$
\end{karte}
